CREATE PROCEDURE [dbo].[GetCustomerById] --'ALFKI'
@CustomerId varchar(15)
AS

@CustomerId varchar(15)
AS 
SELECT [CustomerID]
      ,[CompanyName]
      ,[ContactName]
      ,[ContactTitle]
      ,[Address]
      ,[City]
      ,[Region]
      ,[PostalCode]
      ,[Country]
      ,[Phone]
      ,[Fax]
	   FROM Customer where customerid = @CustomerId 

SELECT [OrderID]
      ,[CustomerID]
      ,[EmployeeID]
      ,[OrderDate]
      ,[RequiredDate]
      ,[ShippedDate]
      ,[ShipVia]
      ,[Freight]
      ,[ShipName]
      ,[ShipAddress]
      ,[ShipCity]
      ,[ShipRegion]
      ,[ShipPostalCode]
      ,[ShipCountry]
	   FROM ORDER where customerid = @CustomerId 

SELECT DT.[OrderID]
      ,DT.[ProductID]
      ,DT.[UnitPrice]
      ,DT.[Quantity]
      ,DT.[Discount]
	  FROM ORDERDETAIL DT INNER JOIN  ORDER O ON DT.ORDERID = O.ORDERID WHERE O.CUSTOMERID = @CustomerId  

SELECT p.[ProductID]
      ,p.[ProductName]
      ,p.[SupplierID]
      ,p.[CategoryID]
      ,p.[QuantityPerUnit]
      ,p.[UnitPrice]
      ,p.[UnitsInStock]
      ,p.[UnitsOnOrder]
      ,p.[ReorderLevel]
	  ,[Discontinued]
	   FROM Product p inner join ORDERDETAIL DT on p.productid = dt.productid INNER JOIN  ORDER O ON DT.ORDERID = O.ORDERID WHERE O.CUSTOMERID = @CustomerId  


SELECT E.[EmployeeID]
      ,E.[LastName]
      ,E.[FirstName]
      ,E.[Title]
      ,E.[TitleOfCourtesy]
      ,E.[BirthDate]
      ,E.[HireDate]
      ,E.[Address]
      ,E.[City]
      ,E.[Region]
      ,E.[PostalCode]
      ,E.[Country]
      ,E.[HomePhone]
      ,E.[Extension]
      ,null as [Photo]
      ,E.[Notes]
      ,E.[ReportsTo]
      ,E.[PhotoPath]
  FROM [dbo].[Employee] E inner
  JOIN  ORDER O ON E.EmployeeID  = O.EmployeeID  WHERE O.CUSTOMERID = @CustomerId

CREATE procedure [dbo].[InsertCustomer]
            @CustomerID nchar(5)
           ,@CompanyName nvarchar(40)
           ,@ContactName nvarchar(30)
           ,@ContactTitle nvarchar(30)
           ,@Address nvarchar(60)
           ,@City nvarchar(15)
           ,@Region nvarchar(15) = NULL
           ,@PostalCode nvarchar(10)
           ,@Country nvarchar(15)
           ,@Phone nvarchar(24)
           ,@Fax nvarchar(24)
as

INSERT INTO [dbo].[Customer]
           ([CustomerID]
           ,[CompanyName]
           ,[ContactName]
           ,[ContactTitle]
           ,[Address]
           ,[City]
           ,[Region]
           ,[PostalCode]
           ,[Country]
           ,[Phone]
           ,[Fax])
     VALUES
           (
		    @CustomerID 
           ,@CompanyName
           ,@ContactName
           ,@ContactTitle
           ,@Address
           ,@City
           ,@Region
           ,@PostalCode
           ,@Country
           ,@Phone
           ,@Fax
		   )

CREATE procedure [dbo].[SaveCustomer] 
            @CustomerID nchar(5)
           ,@CompanyName nvarchar(40)
           ,@ContactName nvarchar(30)
           ,@ContactTitle nvarchar(30)
           ,@Address nvarchar(60)
           ,@City nvarchar(15)
           ,@Region nvarchar(15) = NULL
           ,@PostalCode nvarchar(10)
           ,@Country nvarchar(15)
           ,@Phone nvarchar(24)
           ,@Fax nvarchar(24)
AS



If EXISTS (SELECT NULL FROM [dbo].[Customer] WHERE CustomerID = @CustomerID)
	BEGIN
	EXEC [dbo].[UpdateCustomer]
	       @CustomerID 
           ,@CompanyName 
           ,@ContactName 
           ,@ContactTitle 
           ,@Address 
           ,@City 
           ,@Region 
           ,@PostalCode
           ,@Country 
           ,@Phone 
           ,@Fax 
	END
ELSE
	BEGIN
	EXEC [dbo].[InsertCustomer]
	       @CustomerID 
           ,@CompanyName 
           ,@ContactName 
           ,@ContactTitle 
           ,@Address 
           ,@City 
           ,@Region 
           ,@PostalCode
           ,@Country 
           ,@Phone 
           ,@Fax 
	END 


	EXEC [dbo].[GetCustomerById] @CustomerID 

		
GO

CREATE procedure [dbo].[UpdateCustomer]
            @CustomerID nchar(5)
           ,@CompanyName nvarchar(40)
           ,@ContactName nvarchar(30)
           ,@ContactTitle nvarchar(30)
           ,@Address nvarchar(60)
           ,@City nvarchar(15)
           ,@Region nvarchar(15) = NULL
           ,@PostalCode nvarchar(10)
           ,@Country nvarchar(15)
           ,@Phone nvarchar(24)
           ,@Fax nvarchar(24)
as

UPDATE [dbo].[Customer]
   SET [CustomerID] = @CustomerID
      ,[CompanyName] = @CompanyName
      ,[ContactName] = @ContactName
      ,[ContactTitle] = @ContactTitle
      ,[Address] = @Address
      ,[City] = @City
      ,[Region] = @Region
      ,[PostalCode] = @PostalCode
      ,[Country] = @Country 
      ,[Phone] = @Phone
      ,[Fax] = @Fax
 WHERE  CustomerID = @CustomerID

GO
